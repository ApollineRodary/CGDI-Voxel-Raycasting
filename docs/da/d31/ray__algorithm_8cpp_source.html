12bool rayHitsBox(const Point& origin, const Point& direction, const AABB& box, double& distance) {
13 double t_near = -HUGE_VAL;
14 double t_far = HUGE_VAL;
15
16 // Repeat for every pair of parallel planes
17 for(int axis=0; axis<3; ++axis) {
18 // If the ray is parallel to the planes, check whether the origin lies between them
19 if (direction[axis] == 0) {
20 if (origin[axis] < box.min[axis] || origin[axis] > box.max[axis])
21 return false;
22 continue;
23 }
24
25 // Compute distance to both planes
26 double t1 = (box.min[axis] - origin[axis]) / direction[axis];
27 double t2 = (box.max[axis] - origin[axis]) / direction[axis];
28 if (t1 > t2) {
29 // Swap t1 and t2 if necessary
30 t2 += t1;
31 t1 = t2 - t1;
32 t2 -= t1;
33 }
34 if (t1 > t_near)
35 t_near = t1;
36 if (t2 < t_far)
37 t_far = t2;
38
39 if (t_near > t_far)
40 // Box is missed
41 return false;
42 if (t_far < 0)
43 // Box is behind ray
44 return false;
45 }
46
47 distance = t_near;
48 return true;
49}

51bool SlabAlgorithm::computeStep(Ray& ray, const SandboxScene& scene) {
52 Point prev_point = ray.getLastTracePoint();
53 auto next_tile = VoxelPosition(prev_point + ray.getDirection()*1e-5);
54 auto boxes = scene.getVoxel(next_tile).getContents();
55
56 bool hits_something = false;
57 double min_distance = HUGE_VAL;
58 for (auto& box: boxes) {
59 double distance_to_box;
60 Point origin_relative = prev_point - Point(next_tile.x, next_tile.y, next_tile.z);
61 if (rayHitsBox(origin_relative, ray.getDirection(), box, distance_to_box)) {
62 hits_something = true;
63 if (distance_to_box < min_distance)
64 min_distance = distance_to_box;
65 }
66 }
67
68 if (hits_something) {
69 // Advance to the AABB that is hit
70 Point new_point(prev_point + ray.getDirection()*min_distance);
71 ray.addTrace(new_point);
72 } else {
73 // Advance to the next voxel
74 double distance_to_next_voxel = HUGE_VAL;
75 for (int axis=0; axis<3; ++axis) {
76 // offset: how far along the current axis one should move to change tile
77 double offset = fmod(prev_point[axis], 1);
78 if (offset == 0)
79 offset = 1;
80 else if (ray.getDirection()[axis] < 0)
81 offset = 1 - offset;
82
83 // distance: how far along the ray one should move to move by offset on the current axis
84 double distance = offset / std::abs(ray.getDirection()[axis]);
85 if (distance < distance_to_next_voxel)
86 distance_to_next_voxel = distance;
87 }
88 Point new_point(prev_point + ray.getDirection()*distance_to_next_voxel);
89 ray.addTrace(new_point);
90 }
91
92 return hits_something;
93}
