10std::vector<AABB> str_to_aabbvector(const std::string& s) {
11 unsigned long curr=1;// Skip the first [
12 unsigned long start;
13 double x_min, y_min, z_min, x_max, y_max, z_max;
14 std::vector<AABB> output;
15
16 while (curr<s.length()-1) {
17 curr += 5;// Skip AABB[
18

20 start = curr;

25
26 // Get the y min coord

32

40 // Get the x max coord
